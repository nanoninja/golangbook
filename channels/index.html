Channels

Don’t communicate by sharing memory; share memory by communicating

Rob Pike

package main

import "fmt"

func emit(message chan string) {
    message <- "Hello, World"
}

func print(message chan string) {
    fmt.Println(<-message)
}

func main() {
    msg := make(chan string)

    go emit(msg)

    print(msg)
}
